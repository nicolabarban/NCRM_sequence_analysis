\frame{
\frametitle{Example 2: Family trajectories and Health}
From:\\

Barban (2013) Family Trajectories and Health: A Life Course Perspective. European Journal of Population
\textbf{What is the association between family trajectories and  health?\\
}Lower health outcomes may be  associated with:
\begin{enumerate}
	\item Earlier transitions (\emph{timing})
	\item Number of transitions (\emph{quantum})
	\item Non-normative transitions (\emph{sequencing})
\end{enumerate}
Moreover, there might be some specific \textbf{patterns} of family formation that are associated with lower health outcomes. 

}



\frame{
\frametitle{Data}
\begin{itemize}
	\item National Longitudinal Study of Adolescent Health \textbf{ (Add Health) }
	\item Nationally representative sample of U.S. students in grades 7 through 12 in 1994. Cohort born 1976-1982.
	\item Four waves:  WI 1995; WII 1996; WIII 2001--2002; WIV 2008--2009.  
	\item Sample size:  20,000 students in wave I 	
\end{itemize}
For this study I restrict the sample to women  who are 30 or older at Wave IV. The final sample size is 2,358
}		


\frame{
\frametitle{Sequences' state space}
Monthly sequence from age 15 to 30\\
In each month  individuals can be classified as:\\
\begin{enumerate}
	\item Single (S)
	\item Single Parent (SP)
	\item Cohabiting (C) 
	\item Cohabiting parent (CP) 
	\item Married (M) 
	\item Married parent (MP)
	\end{enumerate}
	
}

\frame{
\frametitle{Data}
\textbf{Data:} Add-health. Women 30 or older. \\
\textbf{Health Outcomes:} (continuous vars)

 \begin{itemize}
	\item Self-reported health
	\item CES--D Depression scale
	\item $\#$ cigarettes smoked in the last month
	\item $\#$ number of episodes of heavy drinking in the last year (5 or more alcoholic cocktails) 
\end{itemize}

}

\frame{
\textbf{Independent variables:}\\ 
\begin{itemize}
	\item Age at first transition (union, child) (\emph{timing})
	\item Number of transitions from wave I to wave IV (\emph{quantum})
	\item Number of non-normative transitions from wave I to wave IV (norm$\rightarrow$ S--M--MP) (\emph{ordering})\\
	
	\textbf{Controls:}\\ 
	\begin{itemize}
		\item Age, Age sq., Race/Ethnicity, Family composition at Wave I, Parent's education.
	\end{itemize}
	
\end{itemize}
}

\frame{
\frametitle{Methods }
Lagged dependent variable model\\
\begin{equation}
	Y_{i2}=\gamma D_i+ \rho Y_{i1}+\beta_i X_{i1} +\epsilon_i2
\end{equation}
where:
\begin{itemize}
	\item $Y_{i2}$ is vector of health indicators measured at Wave IV (Time 2)
	\item $Y_{i1}$ represents a vector of identical health measures at Wave I (Time 1)
	 \item  $D_i$ represents the characteristics of the sequence from Wave I to Wave IV. 
	\item $X_{i1}$  a vector of demographic controls and SES background at Wave I (race; parents' education; )
\end{itemize}
}
